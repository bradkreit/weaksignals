Live from Google Sheets

One-time setup required

Because this dashboard runs as a local HTML file, browsers block direct Google Sheets requests (CORS). The fix is a tiny Google Apps Script — a 10-line function you paste once into your sheet. It acts as a relay, and this only needs to be done once.

1
Open your Google Sheet, then go to Extensions → Apps Script.
2
Delete any existing code in the editor and paste in the following:
function doGet(e) { if (!e || !e.parameter) { return respond({ status: 'ok', message: 'Proxy running.' }); } var action = e.parameter.action || 'sheet'; if (action === 'sheet') { var tab = SpreadsheetApp.getActiveSpreadsheet() .getSheetByName(e.parameter.sheet || 'LiveFeed'); if (!tab) return respond({ error: 'Sheet not found' }); return respond({ sheet: e.parameter.sheet, rows: tab.getDataRange().getValues() }); } if (action === 'readwise') { var token = e.parameter.token; if (!token) return respond({ error: 'No token' }); var page = parseInt(e.parameter.page || '1', 10); var url = 'https://readwise.io/api/v2/export/?page=' + page + '&page_size=100'; var r = UrlFetchApp.fetch(url, { headers: { 'Authorization': 'Token ' + token }, muteHttpExceptions: true }); if (r.getResponseCode() === 401) return respond({ error: 'Invalid Readwise token' }); if (r.getResponseCode() === 429) return respond({ error: 'rate_limited' }); if (r.getResponseCode() !== 200) return respond({ error: 'HTTP ' + r.getResponseCode() }); var d = JSON.parse(r.getContentText()); var books = (d.results || []).map(function(b) { return { id: b.id, title: b.title || 'Untitled', author: b.author || '', category: b.category || 'books', source_url: b.source_url || '', cover_image_url: b.cover_image_url || '', highlights: (b.highlights || []).map(function(h) { return { text: h.text || '', note: h.note || '', url: h.url || '', highlighted_at: h.highlighted_at || '', tags: (h.tags || []).map(function(t) { return t.name || t; }) }; }) }; }); return respond({ books: books, page: page, has_more: !!d.next, count: books.length }); } return respond({ error: 'Unknown action' }); } function respond(data) { return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
3
Click Deploy → New deployment. Set type to Web app. Set "Who has access" to Anyone. Click Deploy and authorize when prompted.
4
Copy the Web app URL (looks like https://script.google.com/macros/s/ABC.../exec) and paste it below.
